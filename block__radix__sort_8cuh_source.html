262  template <int DESCENDING, int KEYS_ONLY>
263  __device__ __forceinline__ void SortBlocked(
264  Key (&keys)[ITEMS_PER_THREAD],
265  Value (&values)[ITEMS_PER_THREAD],
266  int begin_bit,
267  int end_bit,
268  Int2Type<DESCENDING> is_descending,
269  Int2Type<KEYS_ONLY> is_keys_only)
270  {
271  UnsignedBits (&unsigned_keys)[ITEMS_PER_THREAD] =
272  reinterpret_cast<UnsignedBits (&)[ITEMS_PER_THREAD]>(keys);
273 
274  // Twiddle bits if necessary
275  #pragma unroll
276  for (int KEY = 0; KEY < ITEMS_PER_THREAD; KEY++)
277  {
278  unsigned_keys[KEY] = KeyTraits::TwiddleIn(unsigned_keys[KEY]);
279  }
280 
281  // Radix sorting passes
282  while (true)
283  {
284  // Rank the blocked keys
285  int ranks[ITEMS_PER_THREAD];
286  RankKeys(unsigned_keys, ranks, begin_bit, is_descending);
287  begin_bit += RADIX_BITS;
288 
289  __syncthreads();
290 
291  // Exchange keys through shared memory in blocked arrangement
292  BlockExchangeKeys(temp_storage.exchange_keys, linear_tid).ScatterToBlocked(keys, ranks);
293 
294  // Exchange values through shared memory in blocked arrangement
295  ExchangeValues(values, ranks, is_keys_only, Int2Type<true>());
296 
297  // Quit if done
298  if (begin_bit >= end_bit) break;
299 
300  __syncthreads();
301  }
302 
303  // Untwiddle bits if necessary
304  #pragma unroll
305  for (int KEY = 0; KEY < ITEMS_PER_THREAD; KEY++)
306  {
307  unsigned_keys[KEY] = KeyTraits::TwiddleOut(unsigned_keys[KEY]);
308  }
309  }
