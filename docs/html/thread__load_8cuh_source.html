120 template <int COUNT, int MAX>
121 struct IterateThreadLoad
122 {
123  template <CacheLoadModifier MODIFIER, typename T>
124  static __device__ __forceinline__ void Load(T *ptr, T *vals)
125  {
126  vals[COUNT] = ThreadLoad<MODIFIER>(ptr + COUNT);
127  IterateThreadLoad<COUNT + 1, MAX>::template Load<MODIFIER>(ptr, vals);
128  }
129 
130  template <typename InputIterator, typename T>
131  static __device__ __forceinline__ void Dereference(InputIterator ptr, T *vals)
132  {
133  vals[COUNT] = ptr[COUNT];
134  IterateThreadLoad<COUNT + 1, MAX>::Dereference(ptr, vals);
135  }
136 };

400 template <typename T, int MODIFIER>
401 __device__ __forceinline__ T ThreadLoad(
402  T *ptr,
403  Int2Type<MODIFIER> modifier,
404  Int2Type<true> is_pointer)
405 {
406  typedef typename UnitWord<T>::DeviceWord DeviceWord;
407 
408  const int DEVICE_MULTIPLE = sizeof(T) / sizeof(DeviceWord);
409 
410  DeviceWord words[DEVICE_MULTIPLE];

412  IterateThreadLoad<0, DEVICE_MULTIPLE>::template Load<CacheLoadModifier(MODIFIER)>(
413  reinterpret_cast<DeviceWord*>(ptr),
414  words);
