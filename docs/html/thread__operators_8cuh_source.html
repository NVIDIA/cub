217 template <
218  typename ReductionOp,
219  typename ItemOffsetPair>
220 class ReduceBySegmentOp
221 {
222 private:
223 
225  ReductionOp op;
226 
227 public:
228 
230  __host__ __device__ __forceinline__ ReduceBySegmentOp(ReductionOp op) : op(op) {}
231 
233  __host__ __device__ __forceinline__ ItemOffsetPair operator()(
234  const ItemOffsetPair &first,
235  const ItemOffsetPair &second)
236  {
237  // This expression uses less registers and is faster when compiled with Open64
238  ItemOffsetPair retval;
239  retval.offset = first.offset + second.offset;
240  retval.value = (second.offset) ?
241  second.value : // The second partial reduction spans a segment reset, so it's value aggregate becomes the running aggregate
242  op(first.value, second.value); // The second partial reduction does not span a reset, so accumulate both into the running aggregate
243  return retval;
244  }
245 };
