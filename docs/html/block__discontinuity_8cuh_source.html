341  template <
342  int ITEMS_PER_THREAD,
343  typename FlagT,
344  typename FlagOp>
345  __device__ __forceinline__ void FlagHeads(
346  FlagT (&head_flags)[ITEMS_PER_THREAD],
347  T (&input)[ITEMS_PER_THREAD],
348  FlagOp flag_op)
349  {
350  // Share last item
351  temp_storage.last_items[linear_tid] = input[ITEMS_PER_THREAD - 1];
352 
353  __syncthreads();
354 
355  // Set flag for first thread-item
356  head_flags[0] = (linear_tid == 0) ?
357  1 : // First thread
358  ApplyOp<FlagOp>::Flag(
359  flag_op,
360  temp_storage.last_items[linear_tid - 1],
361  input[0],
362  linear_tid * ITEMS_PER_THREAD);
363 
364  // Set head_flags for remaining items
365  Iterate<1, ITEMS_PER_THREAD>::FlagHeads(linear_tid, head_flags, input, flag_op);
366  }
