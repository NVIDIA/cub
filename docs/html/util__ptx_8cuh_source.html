376 template <typename T>
377 __device__ __forceinline__ T ShuffleUp(
378  T input,
379  int src_offset)
380 {
381  enum
382  {
383  SHFL_C = 0,
384  };
385 
386  typedef typename UnitWord<T>::ShuffleWord ShuffleWord;
387 
388  const int WORDS = (sizeof(T) + sizeof(ShuffleWord) - 1) / sizeof(ShuffleWord);
389  T output;
390  ShuffleWord *output_alias = reinterpret_cast<ShuffleWord *>(&output);
391  ShuffleWord *input_alias = reinterpret_cast<ShuffleWord *>(&input);
392 
393  #pragma unroll
394  for (int WORD = 0; WORD < WORDS; ++WORD)
395  {
396  unsigned int shuffle_word = input_alias[WORD];
397  asm(
398  " shfl.up.b32 %0, %1, %2, %3;"
399  : "=r"(shuffle_word) : "r"(shuffle_word), "r"(src_offset), "r"(SHFL_C));
400  output_alias[WORD] = (ShuffleWord) shuffle_word;
401  }
402 
403  return output;
404 }
