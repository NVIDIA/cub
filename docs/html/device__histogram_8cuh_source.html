124 template <
125  int BINS,
126  int ACTIVE_CHANNELS,
127  typename HistoCounter>
128 __launch_bounds__ (BINS, 1)
129 __global__ void HistoAggregateKernel(
130  HistoCounter* d_block_histograms,
131  ArrayWrapper<HistoCounter*, ACTIVE_CHANNELS> d_out_histograms,
132  int num_threadblocks)
133 {
134  // Accumulate threadblock-histograms from the channel
135  HistoCounter bin_aggregate = 0;
136 
137  int block_offset = blockIdx.x * (num_threadblocks * BINS);
138  int block_end = block_offset + (num_threadblocks * BINS);
139 
140 #if CUB_PTX_VERSION >= 200
141  #pragma unroll 32
142 #endif
143  while (block_offset < block_end)
144  {
145  HistoCounter block_bin_count = d_block_histograms[block_offset + threadIdx.x];
146 
147  bin_aggregate += block_bin_count;
148  block_offset += BINS;
149  }
150 
151  // Output
152  d_out_histograms.array[blockIdx.x][threadIdx.x] = bin_aggregate;
153 }

328  template <
329  typename InitHistoKernelPtr,
330  typename HistoRegionKernelPtr,
331  typename AggregateHistoKernelPtr>
332  __host__ __device__ __forceinline__
333  static cudaError_t Dispatch(
334  void *d_temp_storage,
335  size_t &temp_storage_bytes,
336  InputIterator d_samples,
337  HistoCounter *d_histograms[ACTIVE_CHANNELS],
338  Offset num_samples,
339  cudaStream_t stream,
340  bool debug_synchronous,
341  InitHistoKernelPtr init_kernel,
342  HistoRegionKernelPtr histo_region_kernel,
343  AggregateHistoKernelPtr aggregate_kernel,
344  KernelConfig histo_region_config)
345  {
346  #ifndef CUB_RUNTIME_ENABLED
347 
348  // Kernel launch not supported from this device
349  return CubDebug(cudaErrorNotSupported);
350 
351  #else
352 
353  cudaError error = cudaSuccess;
354  do
355  {
356  // Get device ordinal
357  int device_ordinal;
358  if (CubDebug(error = cudaGetDevice(&device_ordinal))) break;
359 
360  // Get device SM version
361  int sm_version;
362  if (CubDebug(error = SmVersion(sm_version, device_ordinal))) break;
363 
364  // Get SM count
365  int sm_count;
366  if (CubDebug(error = cudaDeviceGetAttribute (&sm_count, cudaDevAttrMultiProcessorCount, device_ordinal))) break;
367 
368  // Get SM occupancy for histo_region_kernel
369  int histo_region_sm_occupancy;
370  if (CubDebug(error = MaxSmOccupancy(
371  histo_region_sm_occupancy,
372  sm_version,
373  histo_region_kernel,
374  histo_region_config.block_threads))) break;
375 
376  // Get device occupancy for histo_region_kernel
377  int histo_region_occupancy = histo_region_sm_occupancy * sm_count;
378 
379  // Get tile size for histo_region_kernel
380  int channel_tile_size = histo_region_config.block_threads * histo_region_config.items_per_thread;
381  int tile_size = channel_tile_size * CHANNELS;
382 
383  // Even-share work distribution
384  int subscription_factor = histo_region_sm_occupancy; // Amount of CTAs to oversubscribe the device beyond actively-resident (heuristic)
385  GridEvenShare<Offset> even_share(
386  num_samples,
387  histo_region_occupancy * subscription_factor,
388  tile_size);
389 
390  // Get grid size for histo_region_kernel
391  int histo_region_grid_size;
392  switch (histo_region_config.grid_mapping)
393  {
394  case GRID_MAPPING_EVEN_SHARE:
395 
396  // Work is distributed evenly
397  histo_region_grid_size = even_share.grid_size;
398  break;
399 
400  case GRID_MAPPING_DYNAMIC:
401 
402  // Work is distributed dynamically
403  int num_tiles = (num_samples + tile_size - 1) / tile_size;
404  histo_region_grid_size = (num_tiles < histo_region_occupancy) ?
405  num_tiles : // Not enough to fill the device with threadblocks
406  histo_region_occupancy; // Fill the device with threadblocks
407  break;
408  };
409 
410  // Temporary storage allocation requirements
411  void* allocations[2];
412  size_t allocation_sizes[2] =
413  {
414  ACTIVE_CHANNELS * histo_region_grid_size * sizeof(HistoCounter) * BINS, // bytes needed for privatized histograms
415  GridQueue<int>::AllocationSize() // bytes needed for grid queue descriptor
416  };
417 
418  // Alias the temporary allocations from the single storage blob (or set the necessary size of the blob)
419  if (CubDebug(error = AliasTemporaries(d_temp_storage, temp_storage_bytes, allocations, allocation_sizes))) break;
420  if (d_temp_storage == NULL)
421  {
422  // Return if the caller is simply requesting the size of the storage allocation
423  return cudaSuccess;
424  }
425 
426  // Alias the allocation for the privatized per-block reductions
427  HistoCounter *d_block_histograms = (HistoCounter*) allocations[0];
428 
429  // Alias the allocation for the grid queue descriptor
430  GridQueue<Offset> queue(allocations[1]);
431 
432  // Setup array wrapper for histogram channel output (because we can't pass static arrays as kernel parameters)
433  ArrayWrapper<HistoCounter*, ACTIVE_CHANNELS> d_histo_wrapper;
434  for (int CHANNEL = 0; CHANNEL < ACTIVE_CHANNELS; ++CHANNEL)
435  d_histo_wrapper.array[CHANNEL] = d_histograms[CHANNEL];
436 
437  // Setup array wrapper for temporary histogram channel output (because we can't pass static arrays as kernel parameters)
438  ArrayWrapper<HistoCounter*, ACTIVE_CHANNELS> d_temp_histo_wrapper;
439  for (int CHANNEL = 0; CHANNEL < ACTIVE_CHANNELS; ++CHANNEL)
440  d_temp_histo_wrapper.array[CHANNEL] = d_block_histograms + (CHANNEL * histo_region_grid_size * BINS);
441 
442  // Log init_kernel configuration
443  if (debug_synchronous) CubLog("Invoking init_kernel<<<%d, %d, 0, %lld>>>()\n", ACTIVE_CHANNELS, BINS, (long long) stream);
444 
445  // Invoke init_kernel to initialize counters and queue descriptor
446  init_kernel<<<ACTIVE_CHANNELS, BINS, 0, stream>>>(queue, d_histo_wrapper, num_samples);
447 
448  // Sync the stream if specified
449  if (debug_synchronous && (CubDebug(error = SyncStream(stream)))) break;
450 
451  // Whether we need privatized histograms (i.e., non-global atomics and multi-block)
452  bool privatized_temporaries = (histo_region_grid_size > 1) && (histo_region_config.block_algorithm != DEVICE_HISTO_GLOBAL_ATOMIC);
453 
454  // Log histo_region_kernel configuration
455  if (debug_synchronous) CubLog("Invoking histo_region_kernel<<<%d, %d, 0, %lld>>>(), %d items per thread, %d SM occupancy\n",
456  histo_region_grid_size, histo_region_config.block_threads, (long long) stream, histo_region_config.items_per_thread, histo_region_sm_occupancy);
457 
458  // Invoke histo_region_kernel
459  histo_region_kernel<<<histo_region_grid_size, histo_region_config.block_threads, 0, stream>>>(
460  d_samples,
461  (privatized_temporaries) ?
462  d_temp_histo_wrapper :
463  d_histo_wrapper,
464  num_samples,
465  even_share,
466  queue);
467 
468  // Sync the stream if specified
469  if (debug_synchronous && (CubDebug(error = SyncStream(stream)))) break;
470 
471  // Aggregate privatized block histograms if necessary
472  if (privatized_temporaries)
473  {
474  // Log aggregate_kernel configuration
475  if (debug_synchronous) CubLog("Invoking aggregate_kernel<<<%d, %d, 0, %lld>>>()\n",
476  ACTIVE_CHANNELS, BINS, (long long) stream);
477 
478  // Invoke aggregate_kernel
479  aggregate_kernel<<<ACTIVE_CHANNELS, BINS, 0, stream>>>(
480  d_block_histograms,
481  d_histo_wrapper,
482  histo_region_grid_size);
483 
484  // Sync the stream if specified
485  if (debug_synchronous && (CubDebug(error = SyncStream(stream)))) break;
486  }
487  }
488  while (0);
489 
490  return error;
491 
492  #endif // CUB_RUNTIME_ENABLED
493  }
