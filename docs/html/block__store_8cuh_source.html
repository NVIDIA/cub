135 template <
136  typename T,
137  int ITEMS_PER_THREAD>
138 __device__ __forceinline__ void StoreDirectBlockedVectorized(
139  int linear_tid,
140  T *block_ptr,
141  T (&items)[ITEMS_PER_THREAD])
142 {
143  enum
144  {
145  // Maximum CUDA vector size is 4 elements
146  MAX_VEC_SIZE = CUB_MIN(4, ITEMS_PER_THREAD),
147 
148  // Vector size must be a power of two and an even divisor of the items per thread
149  VEC_SIZE = ((((MAX_VEC_SIZE - 1) & MAX_VEC_SIZE) == 0) && ((ITEMS_PER_THREAD % MAX_VEC_SIZE) == 0)) ?
150  MAX_VEC_SIZE :
151  1,
152 
153  VECTORS_PER_THREAD = ITEMS_PER_THREAD / VEC_SIZE,
154  };
155 
156  // Vector type
157  typedef typename CubVector<T, VEC_SIZE>::Type Vector;
158 
159  // Alias global pointer
160  Vector *block_ptr_vectors = reinterpret_cast<Vector *>(block_ptr);
161 
162  // Alias pointers (use "raw" array here which should get optimized away to prevent conservative PTXAS lmem spilling)
163  Vector raw_vector[VECTORS_PER_THREAD];
164  T *raw_items = reinterpret_cast<T*>(raw_vector);
165 
166  // Copy
167  #pragma unroll
168  for (int ITEM = 0; ITEM < ITEMS_PER_THREAD; ITEM++)
169  {
170  raw_items[ITEM] = items[ITEM];
171  }
172 
173  // Direct-store using vector types
174  StoreDirectBlocked(linear_tid, block_ptr_vectors, raw_vector);
175 }
